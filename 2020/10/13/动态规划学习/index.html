    // 用于缓存钞票
    public static HashMap<Integer, Integer> map = new HashMap<>();

    // 寻找最少的钞票数
    public static int f(int m) {
    // #1 处理边界情况
    if (m < 0) {
    return Integer.MAX_VALUE;
    }
    if (m == 0) {
    return 0;
    }
    // #2 加入缓存
    if(map.containsKey(m)){
    return map.get(m);
    }
    // 取 1 和 5 的最小超票数
    int min = Math.min(f(m - 1), f(m - 5));
    // 取 1 、5、11
    min = Math.min(min, f(m - 11));

    // #3 存储缓存
    map.put(m, min + 1);
    return min + 1;
    }
